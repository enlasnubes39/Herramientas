🎬 Control de Video

⚡ Super Velocidad

Aumenta la velocidad del video a 2.5x o la que quieras.

Consola
document.querySelector('video').playbackRate = 2.5
Marcador
javascript:(function(){document.querySelector('video').playbackRate=prompt('Velocidad:',2.5);})();

📑 Reconstructor de PDF

Captura y libera PDFs protegidos de la ECAM.

Marcador
javascript:(async function(){ const viewer = window.PDFViewerApplication; if (!viewer) { alert("Abre el PDF primero."); return; } const totalPages = viewer.pagesCount; viewer.pdfViewer.currentScaleValue = "2.5"; const win = window.open("", "_blank"); win.document.write('Captura PDF
Cargando...
'); const container = win.document.getElementById('container'); const status = win.document.getElementById('status'); for (let i = 1; i <= totalPages; i++) { status.innerText = `Procesando Página ${i} de ${totalPages}`; viewer.page = i; await new Promise(r => setTimeout(r, 2200)); const canvas = viewer.pdfViewer.getPageView(i - 1).canvas; if (canvas) { const img = win.document.createElement('img'); img.src = canvas.toDataURL('image/jpeg', 0.95); img.className = 'page-img'; container.appendChild(img); } else { i--; await new Promise(r => setTimeout(r, 1000)); } } status.innerHTML = "✅ LISTO! "; win.focus(); })();
